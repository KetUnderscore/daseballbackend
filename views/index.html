Welcome to the DASEBALL API!


/players

Options : /sortType/name

sortType Accepts : total, pitching, batting, running, fielding, battingaverage, onbasepercentage, slugging, ops, era, whip, fans


/playerData

Options : /name/season

season Accepts : 1, 2, 3, 4, 4.1, 4.1, 5, 6


/gamePlayerData

Options : /name/season/gameDay

season Accepts : 6

gameDay Accepts : 1-45, 46-55 if Applicable


/teams

Options : /teamName

teamName Accepts : All Active Team Names (eg. Kansas City Mints)


/season

Options : /seasonNumber

seasonNumber Accepts : 1, 2, 3, 4, 4.1, 4.1, 5, 6


/games

Options : /season/day

seasonNumber Accepts : 1, 2, 3, 4, 4.1, 4.1, 5, 6

day Accepts : 1-45, 46-55 if Applicable


/votes

Options : /type

type Accepts : Announcement, Acknowledgement, Allowance
